it is a very simple problem of two pointer



class Solution {
public:
    int minimumRefill(vector<int>& plants, int capacityA, int capacityB) {
        int initA = capacityA;
        int initB = capacityB;
        int a = 0;
        int b = plants.size()-1;
        int refill = 0;
        while(a<=b){
            if(a==b){
                int large = (capacityA>=capacityB)?capacityA:capacityB;
                if(large<plants[a]){
                    refill++;
                    break;
                }
                break;

            }
            if(plants[a]<=capacityA){
                capacityA -= plants[a];
                a++;
            }
            else{
                refill++;
                capacityA = initA;
                capacityA -= plants[a];
                a++;
            }
            if(plants[b]<=capacityB){
                capacityB -= plants[b];
                b--;
            }
            else{
                refill++;
                capacityB = initB;
                capacityB -= plants[b];
                b--;
            }
        }
        return refill;

    }
};
