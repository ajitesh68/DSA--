class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        int left=1;
        int right= *max_element(piles.begin(), piles.end());
        while(left<=right) {
            int mid = left + (right - left) / 2;
            double sum = 0;
            for (int j = 0; j < piles.size(); j++) { 
                 sum += ceil(piles[j] / (double)mid);
            }
            if (sum <= h) 
                right=mid-1;
            else
               left=mid+1;
        }
        return left;
    }
};
