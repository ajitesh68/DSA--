class Solution 
{
public:
    bool hasPathSum(TreeNode* root, int targetSum) 
    {
        if (root == nullptr) return false;//empty tree
        targetSum -= (root->val);
        
        // accept only at a leaf
        if (root->left == nullptr && root->right == nullptr)
        {
            return (targetSum == 0);
        }
        return (hasPathSum(root->left, targetSum) || hasPathSum(root->right, targetSum));//split tree
    }
};
