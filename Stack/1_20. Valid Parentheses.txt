class Solution {
public:
    bool isValid(string s) {
      stack<char> st;
      if(s.size()==0) return false;
      for(int i=0;i<s.size();i++){
        if((s[i]=='(')||(s[i]=='[')||(s[i]=='{')) 
        {
            st.push(s[i]);
            continue;
        }
        if(st.empty()) return false;
        char head = st.top();
        if((s[i]==')')&& (head== '(' )|| ( s[i]==']' ) && (head=='[') || (s[i]=='}') && (head=='{') ) 
          st.pop();
        else return false;    
          
    return (st.empty());
}
};
